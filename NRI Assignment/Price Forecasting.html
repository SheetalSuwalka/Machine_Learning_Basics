Preparing Training data for time series with Lookback method¶

In [9]:
#Create the training data set
#Create the scaled training set
train_data = scaled_data[0:training_data_len , :]

# generate the input and output sequences
n_lookback = 270  # length of input sequences (lookback period)
n_forecast = 90  # length of output sequences (forecast period)

#Lookback period should be multiple of forecast days

#Split the data into x_train and _train data sets
x_train = []
y_train = []


#iterate from lookback index and till size - forecast days so that we get shape of X & Y consistent
for i in range(n_lookback, len(train_data) - n_forecast + 1):
    #X_train is list of list, list will be of 180 sequencial element, and element is "train_data[1:181,0]"
  x_train.append(train_data[i-n_lookback:i,0])
    #y_train is list of element, element will be train_data[181,181+90]"
  y_train.append(train_data[i:i+n_forecast,0])

# For the sequencial 120 training element, the test element will be 121th element, as it is time-series data
#Below is the sample of the training data

#Convetring list to numpy array
x_train = np.array(x_train)
y_train = np.array(y_train)
